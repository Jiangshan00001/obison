97int lalr1::remove_action_from_rules()

99

101 std::vector<std::vector<std::string> > new_rules;

103 m_actions.push_back("{}");//action0 is nothing action

108 for(unsigned i=0;i<m_is_action.size();++i)

143 m_before_action[i] = 0;
144 m_after_action[i] = 0;
145 m_comp_action[i] = 0;

165 this->m_is_action = new_is_action;
166

619 }
620 }

626int lalr1::generate_closures(std::string mstart)
627{
628 Closure first_state;
629
630 for(unsigned i=0;i<m_rules.size();++i)
631 {
632 if(m_rules[i][0]!=mstart)continue;
633
634 min_state curr;
635 curr.m_rule = i;
636 curr.m_curr_dot_index=0;
637 curr.m_next="";
638 first_state.m_kernel.insert(curr);
639 }
640 first_state = get_closure(first_state);
641 m_closures.push_back(first_state);
642
643 for(unsigned i=0;i<m_closures.size();++i)
644 {
645 if(m_is_debug)
646 {
647 if(i%100==0)
648 {
649 std::cout<<"generate_closures: closures:"<<i<<"\n";
650 }
651 }
652
653 auto &onec = m_closures[i];
655 if(m_jmp.size()>i)continue;
656 std::map<std::string, int > jmp_table;
657
658 std::map<std::string, Closure > shift_jmp;
659 std::map<std::string, std::vector<int> > reduce_jmp;
660 int ret = get_closure_next_token(onec,shift_jmp, reduce_jmp);
662 for(auto it=shift_jmp.begin();it!=shift_jmp.end();++it)
663 {
664 //it->first;
665 auto new_closure = get_closure( it->second);
666 int clo_index = 0;
667 auto clo_it = std::find(m_closures.begin(), m_closures.end(), new_closure);
668 if (clo_it == m_closures.end())
669 {
670 m_closures.push_back(new_closure);
671 clo_index = m_closures.size() - 1;
672 }
673 else
674 {
675 clo_index = clo_it - m_closures.begin();
676 }
677
678 jmp_table[it->first] = clo_index;
679 }
680 m_jmp.push_back(jmp_table);
681 m_reduce.push_back(reduce_jmp);
682 }
683 std::cout<<"closures_cnt:"<<m_closures.size()<<"\n";

685

735int lalr1::calc_skip_token()
736{
737
738

740 m_eps_tokens.clear();
741 m_eps_tokens.push_back(EPS_SYM);
742 while(eps_cnt<m_eps_tokens.size())
743 {
744 eps_cnt=m_eps_tokens.size();
745 for(unsigned i=0;i<m_rules.size();++i)
746 {
747 if(std::find(m_eps_tokens.begin(), m_eps_tokens.end(), m_rules[i][0])!=m_eps_tokens.end())
748 {
749 //已经添加过了，直接下一个
750 continue;
751 }
752
753 int can_skip=1;
754 for(int j=1;j<m_rules[i].size();++j)
755 {
756 if(std::find(m_eps_tokens.begin(), m_eps_tokens.end(), m_rules[i][j])==m_eps_tokens.end())
757 {
758 can_skip=0;
759 break;
760 }
761 }
762 if(can_skip)
763 {
764 m_eps_tokens.push_back(m_rules[i][0]);
765 m_eps_token_rule[m_rules[i][0] ] = i;
766 }
767 }

772 std::cout<<"eps_tokens:\n";
773 for(unsigned i=0;i<m_eps_tokens.size();++i)
774 {
775 std::cout<<m_eps_tokens[i]<<", ";
